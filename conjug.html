// Function to get the question text from the page
function getQuestionText() {
  // Try common selectors for question text - adjust these based on your page
  const selectors = [
    '.question-text',
    '[data-question]',
    'h1',
    'h2',
    '.prompt',
    '[role="heading"]'
  ];
  
  for (let selector of selectors) {
    const element = document.querySelector(selector);
    if (element && element.innerText.trim()) {
      return element.innerText.trim();
    }
  }
  
  return null;
}

// Auto-translate and fill function
function autoTranslate() {
  const input = document.getElementById('answer-input');
  if (!input) {
    console.log('Input field not found');
    return;
  }
  
  const questionText = getQuestionText();
  if (!questionText) {
    console.log('Could not find question text');
    return;
  }
  
  console.log('Question found:', questionText);
  
  // Look up translation (try exact match first, then case-insensitive)
  let translation = translations[questionText];
  if (!translation) {
    const lowerQuestion = questionText.toLowerCase();
    for (let key in translations) {
      if (key.toLowerCase() === lowerQuestion) {
        translation = translations[key];
        break;
      }
    }
  }
  
  if (translation) {
    console.log('Translation found:', translation);
    setTimeout(() => {
      // If multiple options, pick the first one
      const firstOption = translation.split('/')[0];
      input.value = firstOption;
      
      // Trigger events
      input.dispatchEvent(new Event('input', { bubbles: true }));
      input.dispatchEvent(new Event('change', { bubbles: true }));
      
      console.log('Translation filled:', firstOption);
    }, 1);
  } else {
    console.log('No translation found for:', questionText);
  }
}

// Run once immediately
autoTranslate();

// Set up observer to auto-run when page changes
const observer = new MutationObserver(() => {
  autoTranslate();
});

observer.observe(document.body, { 
  childList: true, 
  subtree: true 
});

console.log('Auto-translator active! Will fill answers after 0.5 seconds.');
